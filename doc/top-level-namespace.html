get ('/home') do
    if verify_login(session[:id])
        db = SQLite3::Database.new("db/webshop.db")
        db.results_as_hash = true
        
        products = db.execute("SELECT * FROM products")
        slim(:home, locals:{
            products: products})
    end
end
